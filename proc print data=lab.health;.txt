/* ----------------------------
   4B. Option 2 (recommended): Group-wise median imputation by gender (keeps biological differences)
   ---------------------------- */

proc means data=lab.health_conv nway median noprint;
    class gender;
    var bmi_num;
    output out=work.gender_bmi_median (drop=_TYPE_ _FREQ_) median=median_bmi;

data lab.health_imp_group;
    merge lab.health_conv (in=a) work.gender_bmi_median;
    by gender;
    if a;
    if missing(bmi_num) then do;
        if not missing(median_bmi) then bmi_num = median_bmi;
        else bmi_num = .; /* if median missing (rare), remain missing */

/* ----------------------------
   2A. If BMI is character: convert to numeric and flag non-numeric tokens
   (Run this block when &bmi_type = 2)
   ---------------------------- */

DATA lab.health_conv;
    SET lab.health;

    /* Normalize textual forms */
    bmi_raw = strip(bmi);
    bmi_uc  = upcase(bmi_raw);

    /* Common textual tokens treated as missing */
    if bmi_uc in ('', 'NA', 'N/A', 'UNKNOWN', '?') then bmi_token_missing = 1;
    else bmi_token_missing = 0;

    /* Attempt safe numeric conversion; '??' suppresses repeated invalid warnings */
    bmi_num = input(bmi_raw, ?? best12.);

    /* If conversion failed but token is numeric-like with stray characters,
       detect alpha characters and set to missing (we keep a diagnostic flag) */
    has_alpha = anyalpha(bmi_raw);
    if has_alpha then bmi_num = .;

    /* Final missing indicator for BMI (true if token missing or conversion failed) */
    bmi_missing = (bmi_token_missing = 1) OR missing(bmi_num);

    KEEP id gender age hypertension heart_disease ever_married work_type Residence_type
         avg_glucose_level bmi bmi_raw bmi_uc bmi_num bmi_missing bmi_token_missing has_alpha stroke;
RUN;

/* Create final engineered dataset */
data lab.health_fe_final;
    set lab.health;

    /* Keep original raw variables */
    age_raw = age;
    bmi_raw = bmi;
    glucose_raw = avg_glucose_level;

    /* =======================
       1. Age Group Categories
       ======================= */
    length age_group $10;
    if age_raw < 18 then age_group = 'Child';
    else if 18 <= age_raw < 40 then age_group = 'Young';
    else if 40 <= age_raw < 60 then age_group = 'Middle';
    else if age_raw >= 60 then age_group = 'Senior';
    else age_group = 'Unknown';

    /* =======================
       2. BMI Categories
       ======================= */
    length bmi_cat $12;
    if bmi_raw < 18.5 then bmi_cat = 'Underweight';
    else if 18.5 <= bmi_raw < 25 then bmi_cat = 'Normal';
    else if 25 <= bmi_raw < 30 then bmi_cat = 'Overweight';
    else if bmi_raw >= 30 then bmi_cat = 'Obese';
    else bmi_cat = 'Unknown';

    /* =======================
       3. Glucose Categories
       ======================= */
    length glucose_cat $15;
    if glucose_raw < 100 then glucose_cat = 'Normal';
    else if 100 <= glucose_raw < 140 then glucose_cat = 'Prediabetes';
    else if glucose_raw >= 140 then glucose_cat = 'Diabetes';
    else glucose_cat = 'Unknown';

    /* =======================
       4. Standardization (Z-score)
       ======================= */
    /* Mean and StdDev should be pre-computed using PROC MEANS */
    /* Example placeholders - replace with your dataset values */
    age_mean = 43.23;  age_std = 22.61;
    bmi_mean = 28.37;  bmi_std = 6.98;
    glucose_mean = 106.14; glucose_std = 45.28;

    age_scaled = (age_raw - age_mean) / age_std;
    bmi_scaled = (bmi_raw - bmi_mean) / bmi_std;
    glucose_scaled = (glucose_raw - glucose_mean) / glucose_std;

    /* =======================
       5. Interaction Term
       ======================= */
    age_bmi_interaction = age_scaled * bmi_scaled;

    /* =======================
       6. One-Hot Encoding
       ======================= */
    /* Residence_type */
    if Residence_type = 'Urban' then do;
        res_urban = 1; res_rural = 0;
    end;
    else if Residence_type = 'Rural' then do;
        res_urban = 0; res_rural = 1;
    end;

    /* Work_type */
    work_private = (work_type = 'Private');
    work_self = (work_type = 'Self-employed');
    work_govt = (work_type = 'Govt_job');
    work_children = (work_type = 'children');
    work_never = (work_type = 'Never_worked');

    /* Smoking_status */
    smoke_former = (smoking_status = 'formerly smoked');
    smoke_never = (smoking_status = 'never smoked');
    smoke_smokes = (smoking_status = 'smokes');
    smoke_unknown = (smoking_status = 'Unknown');
run;
